        let currentIdx = 0;
        let score = { A: 0, B: 0, C: 0, D: 0 };

        function startTest() {
            document.getElementById('startPage').classList.add('hidden');
            document.getElementById('questionPage').classList.remove('hidden');
            showQuestion();
        }

        function showQuestion() {
            if (currentIdx >= questions.length) {
                showResult();
                return;
            }

            const q = questions[currentIdx];
            document.getElementById('questionNumber').innerText = `Q. ${(currentIdx + 1).toString().padStart(2, '0')}`;
            document.getElementById('questionText').innerText = q.q;
            
            const optionsDiv = document.getElementById('answerOptions');
            optionsDiv.innerHTML = '';
            
            q.a.forEach((ans, i) => {
                const btn = document.createElement('button');
                btn.className = 'answer-btn pop-in';
                btn.style.animationDelay = `${i * 0.05}s`;
                btn.innerText = ans;
                btn.onclick = () => selectAnswer(i);
                optionsDiv.appendChild(btn);
            });

            document.getElementById('progressFill').style.width = `${((currentIdx) / questions.length) * 100}%`;
        }

        function selectAnswer(ansIdx) {
            const types = ['A', 'B', 'C', 'D'];
            score[types[ansIdx]]++;
            currentIdx++;
            setTimeout(showQuestion, 250);
        }

        function showResult() {
            document.getElementById('questionPage').classList.add('hidden');
            document.getElementById('resultPage').classList.remove('hidden');

            let maxScore = 0;
            let resultKey = 'A';
            for (let key in score) {
                if (score[key] > maxScore) {
                    maxScore = score[key];
                    resultKey = key;
                }
            }

            const res = results[resultKey];
            
            document.getElementById('resultEmoji').innerText = res.emoji;
            document.getElementById('resultTypeName').innerText = res.title;
            document.getElementById('resultDesc').innerText = res.desc;

            const t1 = document.getElementById('traits1');
            t1.innerHTML = '';
            res.traits1.forEach(t => {
                const s = document.createElement('span');
                s.className = 'bubble';
                s.innerText = t;
                t1.appendChild(s);
            });

            const t2 = document.getElementById('traits2');
            t2.innerHTML = '';
            res.traits2.forEach(t => {
                const s = document.createElement('span');
                s.className = 'bubble';
                s.innerText = t;
                t2.appendChild(s);
            });
            
            window.scrollTo({ top: 0, behavior: 'smooth' });
        }

        function restart() {
            currentIdx = 0;
            score = { A: 0, B: 0, C: 0, D: 0 };
            document.getElementById('resultPage').classList.add('hidden');
            document.getElementById('startPage').classList.remove('hidden');
            document.getElementById('progressFill').style.width = '0%';
        }
